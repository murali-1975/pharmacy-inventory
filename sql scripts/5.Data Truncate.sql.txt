-- Disable triggers temporarily to avoid constraint errors
SET session_replication_role = 'replica';
-- Truncate all data tables and reset ID counters

    public.dispensing,
    public.invoice_line_items,
    public.invoice_payments,
    public.invoices,
    public.stock_adjustments,
    public.stock_batches,
    public.medicine_stock,

    public.manufacturers,
    public.supplier_bank_accounts,
    public.supplier_contact_details,
    public.suppliers

-- Re-enable triggers
SET session_replication_role = 'origin';
-- Verification counts (should all be 0)
